-- Creación de Tablas --

CREATE TABLE ciudades(
 ciudadid INTEGER PRIMARY KEY,
 nombreciudad VARCHAR(50) NOT NULL
 )
 
 CREATE TABLE beneficiarios(
 beneficiarioid INTEGER PRIMARY KEY,
 nombre VARCHAR(75) NOT NULL,
 edad NUMBER NOT NULL,
 ciudadid INTEGER,
 FOREIGN KEY (ciudadid) REFERENCES ciudades (ciudadid)
)

CREATE TABLE ayudas(
 ayudaid INTEGER PRIMARY KEY,
 beneficiarioid INTEGER,
 monto NUMBER NOT NULL,
 motivo VARCHAR(25) NOT NULL,
 FOREIGN KEY (beneficiarioid) REFERENCES beneficiarios (beneficiarioid)
)
